void updateDayNightCycle() {
    // Update the day-night factor based on time or rotation angle
    dayNightFactor += dayNightSpeed;

        dayNightFactor -= 1.0f;
    }
}
void setPlanetLighting() {
    // Calculate the lighting parameters based on the day-night factor
    GLfloat ambientColor[] = {0.2f, 0.2f, 0.2f, 1.0f}; // Night ambient color
    GLfloat diffuseColor[] = {1.0f, 1.0f, 1.0f, 1.0f}; // Day diffuse color
    GLfloat specularColor[] = {0.5f, 0.5f, 0.5f, 1.0f}; // Specular color

    GLfloat lightPosition[] = {50.0f * cos(dayNightFactor * 2 * PI), 50.0f * sin(dayNightFactor * 2 * PI), 0.0f, 1.0f};

    // Set lighting parameters
    glLightfv(GL_LIGHT0, GL_AMBIENT, ambientColor);
    glLightfv(GL_LIGHT0, GL_DIFFUSE, diffuseColor);
    glLightfv(GL_LIGHT0, GL_SPECULAR, specularColor);
    glLightfv(GL_LIGHT0, GL_POSITION, lightPosition);
}

void drawStars(int numStars) {
    srand(42); // Seed for random number generation, change as needed
    glPointSize(1.0);
    glBegin(GL_POINTS);
    for (int i = 0; i < numStars; i++) {
        // Generate random RGB values for each star
        float starR = static_cast<float>(rand()) / RAND_MAX;
        float starG = static_cast<float>(rand()) / RAND_MAX;
        float starB = static_cast<float>(rand()) / RAND_MAX;

        glColor4f(starR, starG, starB, 1.0); // Random color with alpha 1.0 (fully opaque)

        float x = rand() % 200 - 100; // Random x coordinate in the range [-100, 100]
        float y = rand() % 200 - 100; // Random y coordinate in the range [-100, 100]

        // Generate a random point size between 1.0 and 2.0
        float pointSize = 1.0 + static_cast<float>(rand()) / RAND_MAX;

        glPointSize(pointSize);
        glVertex2f(x, y);
    }
    glEnd();
}

void mouseWheel(int button, int dir, int x, int y) {
    if (dir > 0) {
        // Scroll up, increase the scaling factor
        scaleFactor += 0.1f;
    } else {
        // Scroll down, decrease the scaling factor
        scaleFactor -= 0.1f;
        //if (scaleFactor < 0.1f) scaleFactor = 0.1f; // Ensure minimum scale
    }

    glutPostRedisplay();
}

void mouseClick(int button, int state, int x, int y){
	if(button == GLUT_RIGHT_BUTTON && state == GLUT_DOWN){
		mouseClicked = true;
	}
	else{
		mouseClicked = false;
	}
	 if (button == 3) { // Scroll up
        if (sunRadius < maxSunRadius)
            sunRadius++;
    } else if (button == 4) { // Scroll down
        if (sunRadius > minSunRadius)
            sunRadius--;
    }

}

void init(void){
    glClearColor(0.0, 0.0, 0.0, 0.0); // Set the clear color to black with alpha 0.0 (fully transparent)
    glMatrixMode(GL_PROJECTION);
    glLoadIdentity();
    glOrtho(-100.0, 100.0, -100.0, 100.0, -1.0, 1.0);
}
void reshape(int width, int height){
    glViewport(0, 0, width, height);
    glMatrixMode(GL_PROJECTION);
    glLoadIdentity();
    float aspectRatio = (float)width / (float)height;
    if (width >= height) {
        glOrtho(-100.0 * aspectRatio, 100.0 * aspectRatio, -100.0, 100.0, -1.0, 1.0);
    } else {
        glOrtho(-100.0, 100.0, -100.0 / aspectRatio, 100.0 / aspectRatio, -1.0, 1.0);
    }
    glMatrixMode(GL_MODELVIEW);
    glLoadIdentity();
}
